\newcount\tabsp
\def\tab#1{%
\tabsp #1\relax%
\multiply\tabsp-65536\relax%
\advance\tabsp\hsize%
\hbox\bgroup\hskip #1 pt%
\hsize\tabsp sp\relax%
\vbox\bgroup%
}
\def\etab{\egroup\egroup}

\newcount\chaptercount
\newcount\sectioncount
\chaptercount0\relax
\def\chapter#1{%
\sectioncount0\relax%
\advance\chaptercount1\relax%
\centerline{\bf\the\chaptercount. \it #1}%
\smallskip%
\hrule\medskip%
}
\def\section#1{%
\advance\sectioncount1\relax%
\medskip%
\tab{20}%
\vbox{\hbox{\bf\the\chaptercount.\the\sectioncount. \it #1}\smallskip\hrule}%
\etab%
\smallskip%
}
